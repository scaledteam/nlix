@persist [P2F Ps P]:table KPerf KState PName:string
@trigger none

    if (clk()) {

            KState++

    elseif (first()) {
        KPerf = 20
        
        interval(0)
    }
}  
else {
    while (perf(KPerf)) {
        # Counter
        if (KState >= Ps:count()) {
            KState = 1
        }
        else {
            KState++
        }
        
        # Program memory
        P = Ps[KState, table]
        
        # Clear if parent killed
        if (!P["parent", table] & P:exists("parent")) {
            P:clear()
            Ps:remove(KState)
            KState--
            continue
        }
        
        # Program name
        PName = P["name", string]
        
        # If exist binary (function)
        if (P2F:exists(PName)) {
            P2F[PName, string]()
        }
        # If program is not valid
        elseif (!P) {
            # If all programs is not valid
            if (!Ps) {
                runOnTick(0)
                break
            }
            # If one program is not valid
            else {
                Ps:remove(KState)
                KState--
            }
        }
        # If wrong name
        else {
            out("Program \"" + PName + "\" is not exist.\n")
            P:clear()
            Ps:remove(KState)
            KState--
        }
    }
}
