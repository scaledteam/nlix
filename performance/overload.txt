@name overload
@persist Overload

Overload = 20

KList["overload", number] = 1

function overload() {
    if (args() < 1) {
        out(Overload + "\n")
    }
    else {
        local Arg = arg(1):toNumber()
        if (Arg > 0) {
            Overload = Arg
        }
        else {
            out("Entered value is incorrect.\n")
        }
    }
    end()
}
