@name perf

P2F["perf", string] = "setperf"

function setperf() {
    if (args() < 1) {
        out(KPerf + "\n")
    }
    else {
        local Arg = arg(1):toNumber()
        if (Arg > 0) {
            KPerf = Arg
        }
        else {
            out("Entered value is incorrect.\n")
        }
    }
    end()
}
