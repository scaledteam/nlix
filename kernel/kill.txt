@name kill

P2F["kill", string] = "kill"

function kill() {
    if (args() < 1) {
        out("Usage: kill [id]\n")
    }
    else {
        local Id = arg(1):toNumber()
        
        if (Id < 0) {
            out("Entered value is incorrect.\n")
        }
        else {
            Ps:removeTable(Id):clear()
        }
    }
    end()
}
