@name plist

KList["plist", number] = 1

function plist() {
    local Counter = P["counter", number]
    local Out = ""

    for (I = 1, Overload) {
        Counter++
        
        if (args() == 0 | KTasks[Counter, table]["name", string] == arg(1)) {
            Out += Counter + ": " + KTasks[Counter, table]["name", string] + "\n"
        }
        
        if (Counter >= (KTasks:count() - 1)) {
            out(Out)
            end()
            return
        }
    }
    
    out(Out)
    P["counter", number] = Counter
}
