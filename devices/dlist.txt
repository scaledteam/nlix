@name dlist

PEnabled["dlist", number] = 1

function dlist() {
    local List = ""
    foreach (DeviceType, DeviceList:array = S["devices", table]) {
        List += DeviceType + ": " + DeviceList:count() + "\n"
        foreach (I, Device:wirelink = DeviceList) {
            List += "  " + Device:entity():id() + "\n"
        }
    }
    out(List)
    end()
}
