@name uptime

P2F["uptime", string] = "uptime"

function uptime() {
    local Uptime = systime() - S["startTime", number]
    if (arg(1) == "server") {
        Uptime = systime()
    }
    out(format("%02d:%02d:%02d", int(Uptime / 60 / 60) % 24, int(Uptime / 60) % 60, int(Uptime) % 60) + "\n")
    end()
}
