@name uptime

if (terminalInit()) {
    local Uptime = systime() - ThisEnvironment["startTime", number]
    terminalPrint(int(Uptime / 60 / 60 / 24) % 30 + " days, " + int(Uptime / 60 / 60) % 24 + ":" + int(Uptime / 60) % 60 + ":" + int(Uptime) % 60 + "\n")
    programKill()
}
