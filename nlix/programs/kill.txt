@name kill

local Program = Programs[varArray("args")[1, string]:toNumber(), table]
if (Program != NULL) {
    if (terminalInit()) {
        terminalPrint("Try kill \"" + Program:type() + "\".\n")
    }
    Program:programKill()
}
else {
    if (terminalInit()) {
        terminalPrint("Program don't launched!.\n")
    }
}
programKill()
