@name ls

if (terminalInit()) {
    switch (stateGet()) {
        case 0,
            afileListLoad(varArray("args")[1, string])
            stateNext()
            break
        
        case 1,
            if (afileListReady()) {
                terminalPrint(afileListRead():concat("\n") + "\n")
                programKill()
            }
            break
    }
}
