@name server-info

if (terminalInit()) {
    terminalPrint(
        "Name: " + hostname() + "\n"
        + format("Uptime: %02d:%02d:%02d\n", int(systime() / 60 / 60) % 24, int(systime() / 60) % 60, int(systime()) % 60)
        + "Players:\n"
    )
    
    local Output = ""
    foreach (I, Player:entity = players()) {
        Output += Player:id() + " " + Player:name() + "\n"
    }
    terminalPrint(Output)
    
    programKill()
}
