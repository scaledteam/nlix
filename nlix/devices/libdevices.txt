@name libdevices

function number wirelink:isValid() {
    return This != nowirelink() & This:entity():isValid()
}

function string wirelink:type() {
    local Name = This:entity():type()
    
    if (Name == "gmod_wire_expression2" | Name == "expadv_gate") {
        Name = This:entity():getName()
    }
    
    return Name
}

function number deviceGet(Type:string) {
    if (S["devices", table]:exists(Type) & !P["device", wirelink]:isValid()) {
        local Devices = S["devices", table][Type, array]
        
        if (Devices:count() > 0) {
            local Device = Devices:shiftWirelink()
            if (Device:isValid()) {
                P["device", wirelink] = Device
                S["devices", table]["P", table]:pushTable(P)
                S["devices", table]["D", table]:pushWirelink(Device)
                
                if (Devices:count() == 0) {
                    S["devices", table]:remove(Type)
                }
                return 1
            }
        }
        
        if (Devices:count() == 0) {
            S["devices", table]:remove(Type)
        }
    }
    return 0
}

function number deviceReturn() {
    local Device = P["device", wirelink]
    if (Device:isValid()) {
        local DeviceType = Device:type()
        
        if (P:exists(DeviceType)) {
            P[DeviceType, array]:unshiftWirelink(Device)
        }
        else {
            P[DeviceType, array] = array(Device)
        }
    }
    P:remove("device")
}
