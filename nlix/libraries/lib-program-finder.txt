@name lib-program-finder

function programFinderWrite() {
    if (!E:exists("programFinder")) {
        E["programFinder", table] = table()
    }
    
    E["programFinder", table][ThisProgramName, table] = P
}

function table programFinderRead(ProgramName:string) {
    if (E["programFinder", table]:exists(ProgramName)) {
        local Program = E["programFinder", table][ProgramName, table]
        if (Program:isValid()) {
            return Program
        }
        else {
            E["programFinder", table]:remove(ProgramName)
            return NULL
        }
    }
    else {
        return NULL
    }
}
