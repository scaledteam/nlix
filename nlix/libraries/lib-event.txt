PROGRAM_TABLES:pushString("events")

function number eventWaitString(EventName:string) {
    if (ThisProgram["events", table]:count() > 0 & ThisProgram["events", table][ThisProgram["events", table]:count(), string] == EventName) {
        ThisProgram["events", table]:popString()
        return 1
    }
    else {
        return 0
    }
}

function number eventsExists() {
    return ThisProgram["events", table]:count() > 0
}

function number table:eventsExists() {
    return This["events", table]:count() > 0
}
