@name lib-driver-terminal

function number initTerminal() {
    local App = findAppByName("driver-terminal")
    if (App != NULL) {
        varWrite("driver-terminal", App)
        return 1
    }
    else {
        return 0
    }
}

function terminalPrint(Text:string) {
    local App = varTable("driver-terminal")
    
    App:sendEvent("print")
    App:sendEvent(Text)
}

function terminalErase(Count) {
    local App = varTable("driver-terminal")
    
    App:sendEvent("erase")
    App:sendEvent(Count)
}

function terminalClear() {
    varTable("driver-terminal"):sendEvent("clear")
}


function terminalPrintFast(Text:string) {
    local App = findAppByName("driver-terminal")
    
    App:sendEvent("print")
    App:sendEvent(Text)
}

function terminalClearFast() {
    findAppByName("driver-terminal"):sendEvent("clear")
}
