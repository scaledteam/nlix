@name lib-driver-keyboard

function table createVirtualKeyboard() {
    local Device = object("virtual-keyboard")
    
    Device["keys", array] = array()
    
    return Device
}

function number keyboardInit() {
    return driverInit("driver-keyboard")
}

function array keyboardGetKeys() {
    return varTable("driver-keyboard"):varArray("keys")
}
