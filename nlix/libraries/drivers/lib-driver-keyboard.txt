@name lib-driver-keyboard

function number initKeyboard() {
    local App = findAppByName("driver-keyboard")
    if (App != NULL) {
        varWrite("driver-keyboard", App)
        return 1
    }
    else {
        return 0
    }
}

function array keyboardGetKeys() {
    return varTable("driver-keyboard"):varArray("keys")
}
