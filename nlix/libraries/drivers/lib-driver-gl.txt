@name lib-driver-gl

function number glInit() {
    return driverInitForProgram("driver-gl")
}

function glTriangle(Pos1:vector, Pos2:vector, Pos3:vector) {
    local Program = P["driver-gl", table]
    
    Program:eventSend("glTriangle")
    Program:eventSend(Pos1)
    Program:eventSend(Pos2)
    Program:eventSend(Pos3)
}

function number glReady() {
    return P["driver-gl", table]["idle", number]
}

function glStart() {
    P["driver-gl", table]:eventSend("glStart")
}

function glColor(Color:vector) {
    P["driver-gl", table]:eventSend("glColor")
    P["driver-gl", table]:eventSend(Color)
}

function glFillColor(Color:vector) {
    P["driver-gl", table]:eventSend("glFillColor")
    P["driver-gl", table]:eventSend(Color)
}

function glPos(Pos:vector) {
    P["driver-gl", table]:eventSend("glPos")
    P["driver-gl", table]:eventSend(Pos)
}


function glAng(Angle:angle) {
    P["driver-gl", table]:eventSend("glAng")
    P["driver-gl", table]:eventSend(Angle)
}
