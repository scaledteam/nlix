@name lib-driver-gl

function number glInit() {
    return driverInitForProgram("driver-gl")
}

function glTriangle(Pos1:vector, Pos2:vector, Pos3:vector) {
    local App = varTable("driver-gl")
    
    App:eventSend("glTriangle")
    App:eventSend(Pos1)
    App:eventSend(Pos2)
    App:eventSend(Pos3)
}

function number glReady() {
    return varTable("driver-gl"):varNumber("idle")
}

function glStart() {
    varTable("driver-gl"):eventSend("glStart")
}

function glColor(Color:vector) {
    local App = varTable("driver-gl")
    
    App:eventSend("glColor")
    App:eventSend(Color)
}

function glFillColor(Color:vector) {
    local App = varTable("driver-gl")
    
    App:eventSend("glFillColor")
    App:eventSend(Color)
}

function glPos(Pos:vector) {
    local App = varTable("driver-gl")
    
    App:eventSend("glPos")
    App:eventSend(Pos)
}


function glAng(Angle:angle) {
    local App = varTable("driver-gl")
    
    App:eventSend("glAng")
    App:eventSend(Angle)
}
