@name lib-driver-digi

function number digiInit() {
    return driverInitForProgram("driver-digi")
}

function digiDrawPixel(Pos:vector2, Color:vector) {
    local Program = P["driver-digi", table]
    
    Program:eventSend("drawPixel")
    Program:eventSend(Pos)
    Program:eventSend(Color)
}

function vector2 digiResolution() {
    return P["driver-digi", table]["resolution", vector2]
}

function entity digiEntity() {
    return P["driver-digi", table]["entity", entity]
}

function number digiReady() {
    return P["driver-digi", table]["idle", number]
}
