@name lib-driver-cursor

function number initCursor() {
    local App = findAppByName("driver-cursor")
    if (App != NULL) {
        varWrite("driver-cursor", App)
        return 1
    }
    else {
        return 0
    }
}

function vector2 cursorGetPos() {
    return varTable("driver-cursor"):varVector2("pos")
}
