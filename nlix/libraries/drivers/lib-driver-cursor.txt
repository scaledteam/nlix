@name lib-driver-cursor

function table createVirtualCursor() {
    local Device = object("virtual-cursor")
    
    Device["pos", vector2] = vec2()
    Device["use", number] = 0
    
    return Device
}

function number cursorInit() {
    return driverInit("driver-cursor")
}

function vector2 cursorGetPos() {
    return varTable("driver-cursor"):varVector2("pos")
}

function number cursorGetUse() {
    return varTable("driver-cursor"):varNumber("use")
}
