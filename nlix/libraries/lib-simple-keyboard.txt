@name lib-simple-keyboard

function number simpleKeyboardInit() {
    eventInit()
    return driverInitForProgram("simple-keyboard")
}

function simpleKeyboardRequestInput() {
    P["simple-keyboard", table]:eventSend("get-input")
    P["simple-keyboard", table]:eventSend(ThisProgram)
}

function number simpleKeyboardWaitInput() {
    return eventWaitString("get-input")
}

function string simpleKeyboardGetInput() {
    return eventGetString()
}
