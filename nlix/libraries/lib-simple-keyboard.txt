@name lib-simple-keyboard

function number initSimpleKeyboard() {
    local App = findAppByName("simple-keyboard")
    if (App != NULL) {
        varWrite("simple-keyboard", App)
        return 1
    }
    else {
        return 0
    }
}

function requestKeyboardInput() {
    local App = varTable("simple-keyboard")
    
    App:sendEvent("get-input")
    App:sendEvent(ThisApp)
}

function number waitKeyboardInput() {
    return waitEvent("get-input")
}

function string getKeyboardInput() {
    return getEventString()
}
