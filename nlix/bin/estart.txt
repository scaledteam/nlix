@name estart

if (args() < 1) {
    out("Usage: estart [egp program name] [driver name]\n")
}
else {
    local DriverName = "egp"
    if (args() >= 2) {
        DriverName = arg(2)
    }
    
    local Egp = exec(DriverName)
    local Program = exec(arg(1))
    
    Egp["in", array] = Program["out", array] = array()
    Egp["out", array] = Program["in", array] = array()
    
    Egp["parent", table] = Program
    Program["parent", table] = Egp
}

end()
