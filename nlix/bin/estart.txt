@name estart

if (arg(1) == "") {
    out("Usage: estart [egp program name]\n")
    end()
}
else {
    local Driver = exec("egp")
    local Program = exec(arg(1))
    
    Driver["in", array] = Program["out", array] = array()
    Driver["out", array] = Program["in", array] = array()
    
    end()
}
