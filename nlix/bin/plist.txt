@name plist

P2F["plist", string] = "plist"

function plist() {
    local Counter = P["counter", number]
    local Out = ""

    for (I = 1, Overload) {
        Counter++
        
        Out += Counter + ": " + Ps[Counter, table]:type() + "\n"
        
        if (Counter >= (Ps:count() - 1)) {
            out(Out)
            end()
            return
        }
    }
}
