@name uptime

if (!APPS_INFO:exists("uptime")) {
    local Info = table()
    
    APPS_INFO["uptime", table] = Info
}

if (app("uptime")) {
    local Uptime = systime() - SysStartTime
    consolePrintFast(int(Uptime / 60 / 60 / 24) % 30 + " days, " + int(Uptime / 60 / 60) % 24 + ":" + int(Uptime / 60) % 60 + ":" + int(Uptime) % 60 + "\n")
    appCloseSelf()
}
