@name kill

local Program = Programs[varArray("args")[1, string]:toNumber(), table]
if (Program != NULL) {
    terminalPrintFast("Try kill \"" + Program["name", string] + "\".\n")
    Program:programKill()
}
else {
    terminalPrintFast("Program don't launched!.\n")
}
programKill()
