@name kill

if (!APPS_INFO:exists("kill")) {
    local Info = table()
    
    APPS_INFO["kill", table] = Info
}

if (app("kill")) {
    local App = Apps[appS("args"):toNumber(), table]
    if (App != NULL) {
        terminalPrintFast("Try kill \"" + App["name", string] + "\".\n")
        App:appClose()
    }
    else {
        terminalPrintFast("App don't launched!.\n")
    }
    appCloseSelf()
}
