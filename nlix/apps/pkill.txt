@name pkill

local AppName = appS("args")
if (appExists(AppName)) {
    local App = findAppByName(AppName)
    if (App != NULL) {
        terminalPrintFast("Try kill \"" + App["name", string] + "\".\n")
        App:appClose()
    }
    else {
        terminalPrintFast("App don't launched!\n")
    }
}
else {
    terminalPrintFast("Error! Wrong app name!\n")
}
appCloseSelf()
