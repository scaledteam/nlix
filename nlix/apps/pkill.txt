@name pkill

local Program = programFindByName(varArray("args")[1, string])
if (Program != NULL) {
    terminalPrintFast("Try kill \"" + Program["name", string] + "\".\n")
    Program:programKill()
}
else {
    terminalPrintFast("Program don't launched!\n")
}
programKill()
