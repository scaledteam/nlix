@name ls

PList["ls", number] = 1

function ls() {
    switch (P["state", number]) {
        case 0,
            flist(fabsPath(arg(1), P["parent", table]["pwd", string]))
            P["state", number] = 1
            break
        
        case 1,
            if (fwait()) {
                out(fin())
                out("\n")
                end()
            }
            break
    }
}
